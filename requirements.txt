# R Package Dependencies
# HRV Biofeedback Decision-Making RCT
# Install with: install.packages(c("package1", "package2", ...))

# Core Statistical Analysis
effsize         # >= 0.8.1   - Effect sizes (Cohen's d)
lme4            # >= 1.1-34  - Linear mixed-effects models
mediation       # >= 4.5.0   - Causal mediation analysis
TOSTER          # >= 0.4.2   - Equivalence testing (TOST)

# Bayesian Analysis
rstan           # >= 2.26.23 - Stan interface for R
brms            # >= 2.20.1  - Bayesian regression models
BayesFactor     # >= 0.9.12  - Bayes factors
HDInterval      # >= 0.2.4   - Highest density intervals
loo             # >= 2.6.0   - Leave-one-out cross-validation
posterior       # >= 1.5.0   - Working with posterior draws
bayesplot       # >= 1.10.0  - Bayesian plotting

# Data Manipulation & Visualization
ggplot2         # >= 3.4.3   - Data visualization
dplyr           # >= 1.1.3   - Data manipulation
tidyr           # >= 1.3.0   - Data reshaping
ggeffects       # >= 1.3.4   - Marginal effects visualization

# Missing Data & Bootstrap
mice            # >= 3.16.0  - Multiple imputation
boot            # >= 1.3-28  - Bootstrap procedures

# Optional but Recommended
emmeans         # >= 1.8.2   - Estimated marginal means
patchwork       # >= 1.1.3   - Combining plots
knitr           # >= 1.44    - Report generation
rmarkdown       # >= 2.25    - R Markdown documents
